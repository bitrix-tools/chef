import { it, describe } from 'mocha';
import { assert } from 'chai';

import { {{name}} } from '../../src/{{inputPath}}';

		const instance = new {{name}}('testName');

		assert.equal(instance.getName(), 'testName');
	});

		const instance = new {{name}}('testName');
